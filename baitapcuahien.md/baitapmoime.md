Kiểm tra tình hình
git status                   # coi file nào sửa, chưa add, chưa commit
git log                      # coi lịch sử commit
git log --oneline            # coi lịch sử gọn, 1 dòng/commit
git diff                     # coi nội dung đã sửa gì
💾 Lưu thay đổi
git add <file>                # đưa 1 file vào staging
git add .                    # đưa hết file thay đổi vào staging
git commit -m "message"      # lưu snapshot, kèm ghi chú
